#include 
using namespace std;
ifstream cin ("nrapprime.in");
ofstream cout("nrapprime.out");
int prim(int n)
{
    if(n < 2)
        return 0;
    if(n%2==0 && n>2)
        return 0;
    for(int d=3;d*d<=n;d+=2)
        if(n%d==0)
            return 0;
    return 1;
}
int main()
{
    int n,x[100];
    cin>>n;
    for(int i=0;i< n;i++)
        cin>>x[i];
    int cnt=0;
    for(int i=0;i< n;++i)
        if(prim(x[i]))
            cnt++;
    cout<< cnt<< endl;
    return 0;
}
