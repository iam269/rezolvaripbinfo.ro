int i_prim(int n){
    int p1=0,p2=0;
    for(int i=n;i>=2 && p1==0;i--)
    {
        int ok=1;
        if(i%2==0 && i!=2)
            ok=0;
        for(int d=3;d*d<=i;d+=2)
            if(i%d==0)
                ok=0;
        if(ok==1)
            p1=i;
    }
    for(int i=n;p2==0;i++)
    {
        int ok=1;
        if(i%2==0)
            ok=0;
        for(int d=3;d*d<=i;d+=2)
            if(i%d==0)
                ok=0;
        if(ok==1)
            p2=i;
    }
    return p2-p1;

}
