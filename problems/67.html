int a_prim(int n){
    int p=0;
    if(n<=1)
        return 0;
    for(int i=2;i*i<=n && p==0;i++)
        if(n%i==0)
            p=i;
        if(p==0)
            return 0;
        int k=n/p;
        if(k==p)
            return 0;
        if(k%2==0)
            return 0;
        for(int d=3;d*d<=k;d+=2)
            if(k%d==0)
                return 0;
        return 1;
}
