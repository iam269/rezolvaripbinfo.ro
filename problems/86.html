Soluția în C++

#include 

using namespace std;

ifstream cin("halfsort.in");
ofstream cout("halfsort.out");

int main()
{
    int n;
    cin >> n;
    int a[1001];
    for(int i = 1; i <= n; ++i)
        cin >> a[i];

    for(int i = 1; i < n/2; ++i)
        for(int j = i + 1; j <= n/2; ++j)
            if(a[i] > a[j])
                swap(a[i], a[j]);

    for(int i = n/2 + 1; i < n; ++i)
        for(int j = i + 1; j <= n; ++j)
            if(a[i] < a[j])
                swap(a[i], a[j]);

    for(int i = 1; i <= n; ++i)
        cout << a[i] << " ";

    return 0;
}
