using namespace std;
int main()
{
    char s[260],aux[260];
    int n,lmax=0;
    ifstream fin("lgmax.in");
    ofstream fout("lgmax.out");
    fin>>n;
    fin.get();
    while(n){
        fin.getline(s,260);
        if(strlen(s)>lmax)
        {
            lmax=strlen(s);
            strcpy(aux,s);
        }
        n--;
    }
    fout<< aux;
    return 0;
}
