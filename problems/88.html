#include 
using namespace std;
ifstream cin ("palindrom.in");
ofstream cout("palindrom.out");
int main()
{
    int n;
    cin>>n;
    char a[21];
    for(int l=1;l<=n;++l)
    {
        cin>>a;
        int lun=0;
        while(a[lun])
            lun++;
        lun--;
        bool ok=true;
        for(int i=0;i<=lun;++i,--lun)
            if(a[i]!=a[lun])
                ok=false;
        if(ok)
            cout<<"1\n";
        else
            cout<<"0\n";
    }
    return 0;
}
