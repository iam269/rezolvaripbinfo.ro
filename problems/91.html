#include 
using namespace std;
ifstream cin("masini.in");
ofstream cout("masini.out");
int n,t;
int a[1001];

void sortare(int a[],int n){
    bool este_sortat=false;
    while(!este_sortat){
        este_sortat=true;
        for(int i=1;i< n;++i)
            if(a[i]>a[i+1]){
                este_sortat=false;
                swap(a[i],a[i+1]);
            }
    }
}

int main()
{
    cin>>n>>t;
    for(int i=1;i<=n;++i)
        cin>>a[i];
    sortare(a,n);
    int i=1;
    while(t-a[i]>=0 && i<=n)
        t-=a[i],i++;
        cout<< i-1;
    return 0;
}
