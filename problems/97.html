Soluția în C++

#include 
using namespace std;
int f[150];
int main()
{
    char a[21],b[21];
    cin>>a>>b;
    int i=0;
    while(a[i])
    {
        f[a[i]]++;
        i++;
    }
    i=0;
    while(b[i])
    {
        f[b[i]]--;
        i++;
    }
    bool ok=true;
    for(int i=50;i<='z';i++)
        if(f[i]!=0)
            ok=false;
    if(ok)
        cout<< 1;
    else
        cout<< 0;
    return 0;
}
